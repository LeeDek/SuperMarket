DELIMITER //

CREATE PROCEDURE insert_delivery_tree_days()
BEGIN
    DECLARE i INT DEFAULT 0;
    DECLARE delivery_date DATE;
    
    -- Calculate the latest date in the table
    SELECT MAX(delivery_finish_date) INTO delivery_date FROM deliveries;
    
    -- Loop for three days
    WHILE i < 4 DO
        -- Insert four lines for each day
        INSERT INTO deliveries (delivery_finish_date, delivery_start_time,status)
        VALUES
            (DATE_ADD(delivery_date, INTERVAL i DAY), '10:00',1),
            (DATE_ADD(delivery_date, INTERVAL i DAY), '12:00',1),
            (DATE_ADD(delivery_date, INTERVAL i DAY), '14:00',1),
            (DATE_ADD(delivery_date, INTERVAL i DAY), '16:00',1);
            
        SET i = i + 1;
    END WHILE;
END //

DELIMITER ;

	CALL insert_delivery_tree_days();
